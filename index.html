    🧬 Moléculas / Card Header

    Card Header

    Header de tarjeta con fondo coloreado por estado. El color de texto se calcula automáticamente para garantizar contraste WCAG AA.

    CSSJSWCAG AA
    Preview
    Unidad 042Activo
    Unidad 017Advertencia
    Unidad 099Peligro
    Unidad 003Inactivo
    HTML 
    <div class="vsm-card-header" style="background: #16a34a; color: #fff;">
  <span class="vsm-card-title">Unidad 042</span>
  <span class="vsm-status-badge"
        style="--badge-bg: rgba(255,255,255,.2);
               --badge-text: #fff;
               --badge-border: rgba(255,255,255,.3);">
    Activo
  </span>
</div>
    JS 
    // Calcular color de texto con contraste WCAG AA
function getContrastTextColor(hex) {
  const r = parseInt(hex.slice(1,3), 16);
  const g = parseInt(hex.slice(3,5), 16);
  const b = parseInt(hex.slice(5,7), 16);
  const luminance = (0.299*r + 0.587*g + 0.114*b) / 255;
  return luminance > 0.5 ? '#1e293b' : '#ffffff';
}

// Uso
const textColor = getContrastTextColor(vehicle.hexColor);
header.style.background = vehicle.hexColor;
header.style.color = textColor;
